#run a linear regression betw acc and weight
accceleration_weight_model = smf.ols(formula = 'acceleration ~ weight', data = features)
results = accceleration_weight_model.fit()

#extract intercept and coefficient
beta_list = results.params
print(beta_list)
intercept = beta_list[0] 
slope = beta_list[1]

#predicted values of acceleration
acc_estimated = intercept + slope * features["weight"]

#plot scatter plot of weight vs. acc and the fitted line
plt.scatter(x = features["weight"], y = features["acceleration"])
plt.plot(features["weight"], acc_estimated, color = 'red')
plt.legend(labels = ["Data Points", "Line of Best Fit"])
plt.xlabel("Weight")
plt.ylabel("Acceleration")
plt.title("Weight vs. Acceleration")
